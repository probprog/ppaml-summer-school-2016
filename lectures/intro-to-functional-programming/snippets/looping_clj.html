(defn factorial [n]
  "computes n * (n-1) * ... * 1"
  (loop [result 1
         ivals (range 2 (+ n 1))]
    (if (seq ivals)
      (recur (* result (first ivals))
             (rest ivals))
      result)))

(defn factorial [n]
  "computes n * (n-1) * ... * 1"
  (reduce ;; inputs: current result and next value 
          (fn [result i]
            ;; output: result
            (* result i))
          1
          (range (+ n 1))))
